\subsection*{Die Vieldeutigkeit der Namen von Lebewesen}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%
\subsubsection*{\textit{soweli}}
%
\index{\textit{soweli}}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%
\textit{soweli} bedeutet Säugetier. 
\textit{soweli} wird aber auch für das Fleisch von Säugetieren verwendet, da es kein spezielles Wort für Fleisch gibt.

\begin{supertabular}{p{5,5cm}|ll}
soweli ni li ' pona moku. && Das Kalbfleisch schmeckt gut. \\
\end{supertabular} 
%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\subsubsection*{\textit{waso}}
%
\index{\textit{waso}}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%
\textit{waso} schließt alle Vögel und anderen fliegenden Tiere ein. 

\begin{supertabular}{p{5,5cm}|ll}
mi wile moku e waso. && Ich will Hühnchen essen. \\
\end{supertabular} 
%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%
\subsubsection*{\textit{akesi}}
%
\index{\textit{akesi}}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%
\textit{akesi} bezeichnet alle Arten von Reptilien und Amphibien. 
% Viele Monster sind auch \textit{akesi}.

\begin{supertabular}{p{5,5cm}|ll}
akesi	&& Gecko, Echse, Schlange, Dinosaurier, Schildkröte. \\
akesi pi telo moli && Giftiges Reptil, Giftschlange, giftiger Frosch \\
\end{supertabular}
